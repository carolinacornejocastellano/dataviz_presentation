ggplot(df)+
    geom_linerange(aes(
      x = generation,
      ymin = 0, 
      ymax = dollars, 
      colour = reorder(variables, desc(dollars))), 
      position = position_dodge(0.8))+
    geom_point(
      aes(
        x = generation,
        y = dollars,
        colour = reorder(variables, desc(dollars)),
      ),
      size = 4,
      position = position_dodge(0.8)
    ) +
    scale_x_discrete(
    limits = gen_order,
    labels = c("**Silent**

    title =
